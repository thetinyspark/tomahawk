(function() {	
	
	/**

	function Person(){}
	
	/**
	* @method repeat
	* @memberOf tomahawk_ns.Person.prototype
	* @param {string} str The string to repeat.
	* @param {number} How many times to repeat the string.
	**/
	Person.prototype.repeat = function(str, times) {
		if (times === undefined || times < 1) {
			times = 1;
		}
		return new Array(times+1).join(str);
	};
	
	tomahawk_ns.Person = Person;
})()
